58  /***
59   * Deploys the generated site using <code>scp</code> or <code>file</code>
60   * protocol to the site URL specified in the
61   * <code>&lt;distributionManagement&gt;</code> section of the POM.
62   * <p>
63   * For <code>scp</code> protocol, the website files are packaged into zip archive,
64   * then the archive is transfered to the remote host, next it is un-archived.
65   * This method of deployment should normally be much faster
66   * than making a file by file copy.  For <code>file</code> protocol, the files are copied
67   * directly to the destination directory.
68   * </p>
69   *
70   * @author <a href="mailto:michal@org.codehaus.org">Michal Maczka</a>
71   * @phase("deploy")

163     public void execute()
164             throws MojoExecutionException {
165         if (!inputDirectory.exists()) {
166             throw new MojoExecutionException("The site does not exist, please run site:site first");
167         }
168 
169 /*
170         DistributionManagement distributionManagement = project.getDistributionManagement();
171 
172         if ( distributionManagement == null )
173         {
174             throw new MojoExecutionException( "Missing distribution management information in the project" );
175         }
176 
177         Site site = distributionManagement.getSite();
178 
179         if ( site == null )
180         {
181             throw new MojoExecutionException(
182                 "Missing site information in the distribution management element in the project.." );
183         }
184 
185         String url = site.getUrl();
186 
187         String id = site.getId();
188 
189 */
190 
191         String url = remoteServerUrl;
192         String id = remoteServerId;
193 
194         if (id == null) {
195             throw new MojoExecutionException("The remoteServerId is missing in the plugin configuration.");
196         }
197         if (url == null) {
198             throw new MojoExecutionException("The remoteServerUrl is missing in the plugin configuration.");
199         }
200         getLog().debug("The site will be deployed to '" + url + "' with id '" + id + "'");
201 
202         Repository repository = new Repository(id, url);
203 
204         // TODO: work on moving this into the deployer like the other deploy methods
205 
206         Wagon wagon;
207 
208         try {
209             wagon = wagonManager.getWagon(repository);
210             configureWagon(wagon, repository.getId(), settings, container, getLog());
211         } catch (UnsupportedProtocolException e) {
212             throw new MojoExecutionException("Unsupported protocol: '" + repository.getProtocol() + "'", e);
213         } catch (WagonConfigurationException e) {
214             throw new MojoExecutionException("Unable to configure Wagon: '" + repository.getProtocol() + "'", e);
215         }
216 
217         if (!wagon.supportsDirectoryCopy()) {
218             throw new MojoExecutionException(
219                     "Wagon protocol '" + repository.getProtocol() + "' doesn't support directory copying");
220         }
221 
222         try {
223             Debug debug = new Debug();
224 
225             wagon.addSessionListener(debug);
226 
227             wagon.addTransferListener(debug);
228 
229             ProxyInfo proxyInfo = getProxyInfo(repository, wagonManager);
230             if (proxyInfo != null) {
231                 wagon.connect(repository, wagonManager.getAuthenticationInfo(id), proxyInfo);
232             } else {
233                 wagon.connect(repository, wagonManager.getAuthenticationInfo(id));
234             }
235 
236             wagon.putDirectory(inputDirectory, remoteDirectory);
237 
238             if (chmod && wagon instanceof CommandExecutor) {
239                 CommandExecutor exec = (CommandExecutor) wagon;
240                 exec.executeCommand("chmod " + chmodOptions + " " + chmodMode + " " + repository.getBasedir());
241             }
242         } catch (ResourceDoesNotExistException e) {
243             throw new MojoExecutionException("Error uploading site", e);
244         } catch (TransferFailedException e) {
245             throw new MojoExecutionException("Error uploading site", e);
246         } catch (AuthorizationException e) {
247             throw new MojoExecutionException("Error uploading site", e);
248         } catch (ConnectionException e) {
249             throw new MojoExecutionException("Error uploading site", e);
250         } catch (AuthenticationException e) {
251             throw new MojoExecutionException("Error uploading site", e);
252         } catch (CommandExecutionException e) {
253             throw new MojoExecutionException("Error uploading site", e);
254         } finally {
255             try {
256                 wagon.disconnect();
257             } catch (ConnectionException e) {
258                 getLog().error("Error disconnecting wagon - ignored", e);
259             }
260         }
261     }

263     /***
264      * <p>
265      * Get the <code>ProxyInfo</code> of the proxy associated with the <code>host</code>
266      * and the <code>protocol</code> of the given <code>repository</code>.
267      * </p>
268      * <p>
269      * Extract from <a href="http://java.sun.com/j2se/1.5.0/docs/guide/net/properties.html">
270      * J2SE Doc : Networking Properties - nonProxyHosts</a> : "The value can be a list of hosts,
271      * each separated by a |, and in addition a wildcard character (*) can be used for matching"
272      * </p>
273      * <p>
274      * Defensively support for comma (",") and semi colon (";") in addition to pipe ("|") as separator.
275      * </p>
276      *
277      * @param repository   the Repository to extract the ProxyInfo from.
278      * @param wagonManager the WagonManager used to connect to the Repository.
279      * @return a ProxyInfo object instantiated or <code>null</code> if no matching proxy is found
280      */
281     public static ProxyInfo getProxyInfo(Repository repository, WagonManager wagonManager) {
282         ProxyInfo proxyInfo = wagonManager.getProxy(repository.getProtocol());
283 
284         if (proxyInfo == null) {
285             return null;
286         }
287 
288         String host = repository.getHost();
289         String nonProxyHostsAsString = proxyInfo.getNonProxyHosts();
290         String[] nonProxyHosts = StringUtils.split(nonProxyHostsAsString, ",;|");
291         for (int i = 0; i < nonProxyHosts.length; i++) {
292             String nonProxyHost = nonProxyHosts[i];
293             if (StringUtils.contains(nonProxyHost, "*")) {
294                 // Handle wildcard at the end, beginning or middle of the nonProxyHost
295                 String nonProxyHostPrefix = StringUtils.substringBefore(nonProxyHost, "*");
296                 String nonProxyHostSuffix = StringUtils.substringAfter(nonProxyHost, "*");
297                 // prefix*
298                 if (StringUtils.isNotEmpty(nonProxyHostPrefix) && host.startsWith(nonProxyHostPrefix)
299                         && StringUtils.isEmpty(nonProxyHostSuffix)) {
300                     return null;
301                 }
302                 // *suffix
303                 if (StringUtils.isEmpty(nonProxyHostPrefix)
304                         && StringUtils.isNotEmpty(nonProxyHostSuffix) && host.endsWith(nonProxyHostSuffix)) {
305                     return null;
306                 }
307                 // prefix*suffix
308                 if (StringUtils.isNotEmpty(nonProxyHostPrefix) && host.startsWith(nonProxyHostPrefix)
309                         && StringUtils.isNotEmpty(nonProxyHostSuffix) && host.endsWith(nonProxyHostSuffix)) {
310                     return null;
311                 }
312             } else if (host.equals(nonProxyHost)) {
313                 return null;
314             }
315         }
316         return proxyInfo;
317     }

319     /***
320      * Configure the Wagon with the information from serverConfigurationMap ( which comes from settings.xml )
321      *
322      * @param wagon
323      * @param repositoryId
324      * @param settings
325      * @param container
326      * @param log
327      * @throws WagonConfigurationException
328      * @todo Remove when {@link WagonManager#getWagon(Repository) is available}. It's available in Maven 2.0.5.
329      */
330     static void configureWagon(Wagon wagon, String repositoryId, Settings settings, PlexusContainer container,
331                                Log log)
332             throws WagonConfigurationException {
333         // MSITE-25: Make sure that the server settings are inserted
334         for (int i = 0; i < settings.getServers().size(); i++) {
335             Server server = (Server) settings.getServers().get(i);
336             String id = server.getId();
337             if (id != null && id.equals(repositoryId)) {
338                 if (server.getConfiguration() != null) {
339                     final PlexusConfiguration plexusConf =
340                             new XmlPlexusConfiguration((Xpp3Dom) server.getConfiguration());
341 
342                     ComponentConfigurator componentConfigurator = null;
343                     try {
344                         componentConfigurator = (ComponentConfigurator) container.lookup(ComponentConfigurator.ROLE);
345                         componentConfigurator.configureComponent(wagon, plexusConf, container.getContainerRealm());
346                     } catch (final ComponentLookupException e) {
347                         throw new WagonConfigurationException(repositoryId, "Unable to lookup wagon configurator."
348                                 + " Wagon configuration cannot be applied.", e);
349                     } catch (ComponentConfigurationException e) {
350                         throw new WagonConfigurationException(repositoryId, "Unable to apply wagon configuration.",
351                                 e);
352                     } finally {
353                         if (componentConfigurator != null) {
354                             try {
355                                 container.release(componentConfigurator);
356                             } catch (ComponentLifecycleException e) {
357                                 log.error("Problem releasing configurator - ignoring: " + e.getMessage());
358                             }
359                         }
360                     }
361 
362                 }
363 
364             }
365         }
366     }
