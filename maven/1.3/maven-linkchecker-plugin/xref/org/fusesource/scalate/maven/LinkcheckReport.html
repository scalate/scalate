322     /*** {@inheritDoc} */
323     public void execute()
324         throws MojoExecutionException
325     {
326         if ( !canGenerateReport() )
327         {
328             return;
329         }
330 
331         // encoding
332         if ( StringUtils.isEmpty( encoding ) )
333         {
334             if ( getLog().isWarnEnabled() )
335             {
336                 getLog().warn(
337                                "File encoding has not been set, using platform encoding "
338                                    + ReaderFactory.FILE_ENCODING + ", i.e. build is platform dependent!" );
339             }
340             encoding = ReaderFactory.FILE_ENCODING;
341         }
342 
343         List documents = null;
344         try
345         {
346             documents = FileUtils.getFiles( basedir, "**/*.html", null );
347         }
348         catch ( IOException e )
349         {
350             String msg = "IOException: " + e.getMessage();
351             if ( getLog().isDebugEnabled() )
352             {
353                 getLog().error( msg, e );
354             }
355             else
356             {
357                 getLog().error( msg );
358             }
359         }
360 
361         // if the site was not already generated, invoke it
362         if ( documents == null || ( documents != null && documents.size() == 0 ) )
363         {
364             throw new MojoExecutionException( "No site generated yet. Please invoke the scalate:sitegen plugin" );
365         }
366 
367         try
368         {
369             result = executeLinkCheck( basedir );
370         }
371         catch ( LinkCheckException e )
372         {
373             throw new MojoExecutionException( "LinkCheckException: " + e.getMessage(), e );
374         }
375     }

399     /*** {@inheritDoc} */
400     protected void executeReport( Locale locale )
401         throws MavenReportException
402     {
403         if ( result == null )
404         {
405             getLog().debug( "Calling execute()" );
406 
407             try
408             {
409                 this.execute();
410             }
411             catch ( MojoExecutionException e )
412             {
413                 throw new MavenReportException( "MojoExecutionException: " + e.getMessage(), e );
414             }
415         }
416 
417         if ( result != null )
418         {
419             generateReport( locale, result );
420             // free memory
421             result = null;
422         }
423     }

429     /***
430      * Execute the <code>Linkcheck</code> tool.
431      *
432      * @param basedir not null
433      * @throws LinkCheckException if any
434      */
435     private LinkcheckModel executeLinkCheck( File basedir )
436         throws LinkCheckException
437     {
438         // Wrap linkcheck
439         linkCheck.setOnline( !offline );
440         linkCheck.setBasedir( basedir );
441         linkCheck.setBaseURL( baseURL );
442         linkCheck.setReportOutput( linkcheckOutput );
443         linkCheck.setLinkCheckCache( linkcheckCache );
444         linkCheck.setExcludedLinks( excludedLinks );
445         linkCheck.setExcludedPages( getExcludedPages() );
446         linkCheck.setExcludedHttpStatusErrors( excludedHttpStatusErrors );
447         linkCheck.setExcludedHttpStatusWarnings( excludedHttpStatusWarnings );
448         linkCheck.setEncoding( ( StringUtils.isNotEmpty( encoding ) ? encoding : WriterFactory.UTF_8 ) );
449 
450         HttpBean bean = new HttpBean();
451         bean.setMethod( httpMethod );
452         bean.setFollowRedirects( httpFollowRedirect );
453         bean.setTimeout( timeout );
454         if ( httpClientParameters != null )
455         {
456             bean.setHttpClientParameters( httpClientParameters );
457         }
458 
459         Proxy proxy = settings.getActiveProxy();
460         if ( proxy != null )
461         {
462             bean.setProxyHost( proxy.getHost() );
463             bean.setProxyPort( proxy.getPort() );
464             bean.setProxyUser( proxy.getUsername() );
465             bean.setProxyPassword( proxy.getPassword() );
466         }
467         linkCheck.setHttp( bean );
468 
469         return linkCheck.execute();
470     }

472     /***
473      * @return the excludedPages defined by the user and also this report.
474      */
475     private String[] getExcludedPages()
476     {
477         List pagesToExclude =
478             ( excludedPages != null ? new ArrayList( Arrays.asList( excludedPages ) ) : new ArrayList() );
479 
480         // Exclude this report
481         pagesToExclude.add( getOutputName() + ".html" );
482 
483         return (String[]) pagesToExclude.toArray( new String[0] );
484     }

514             getSink().body_();

524         getSink().text( getName( locale ) );
525         getSink().sectionTitle1_();

530

532

534         generateSummarySection( locale, linkcheckModel );
535 
536         if ( linkcheckModel.getFiles().size() > 0 )

540         }
541 
542         getSink().body_();
543         getSink().flush();
544         getSink().close();
545 
546         closeReport();
547     }

549     private void generateSummarySection( Locale locale, LinkcheckModel linkcheckModel )
550     {
551         // Calculus
552         List linkcheckFiles = linkcheckModel.getFiles();
553 
554         int totalFiles = linkcheckFiles.size();
555 
556         int totalLinks = 0;
557         int totalValidLinks = 0;
558         int totalErrorLinks = 0;
559         int totalWarningLinks = 0;
560         for ( Iterator it = linkcheckFiles.iterator(); it.hasNext(); )
561         {
562             LinkcheckFile linkcheckFile = (LinkcheckFile) it.next();
563 
564             totalLinks += linkcheckFile.getNumberOfLinks();
565             totalValidLinks += linkcheckFile.getNumberOfLinks( LinkcheckFileResult.VALID_LEVEL );
566             totalErrorLinks += linkcheckFile.getNumberOfLinks( LinkcheckFileResult.ERROR_LEVEL );

966     /***
967      * Similar to {@link Arrays#toString(int[])} in 1.5.
968      *
969      * @param a not null
970      * @return the array comma separated.
971      */
972     private static String toString( int[] a )
973     {
974         if ( a == null || a.length == 0 )
975         {
976             return "";
977         }
978 
979         StringBuffer buf = new StringBuffer();
980         buf.append( a[0] );
981 
982         for ( int i = 1; i < a.length; i++ )
983         {
984             buf.append( ", " );
985             buf.append( a[i] );
986         }
987 
988         return buf.toString();
989     }
